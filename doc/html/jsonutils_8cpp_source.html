48  void JsonParser::jsonToMap(json_t * json, VariantMap & map)
49  {
50  void * iter = json_object_iter(json);
51  while (iter)
52  {
53  const char * k = json_object_iter_key(iter);
54  json_t * o = json_object_iter_value(iter);
55  if (k && o)
56  {
57  switch (json_typeof(o))
58  {
59  case JSON_OBJECT:
60  {
61  VariantMap m;
62  JsonParser::jsonToMap(o, m);
63  map[k] = m;
64  }
65  break;
66 
67  case JSON_ARRAY:
68  {
69  VariantList l;
70  JsonParser::jsonToList(o, l);
71  map[k] = l;
72  }
73  break;
74 
75  case JSON_STRING: map[k] = REString(json_string_value(o)); break;
76  case JSON_INTEGER: map[k] = Variant((long long)json_integer_value(o)); break;
77  case JSON_REAL: map[k] = Variant((double)json_real_value(o)); break;
78  case JSON_TRUE: map[k] = Variant(true); break;
79  case JSON_FALSE: map[k] = Variant(false); break;
80  case JSON_NULL: map[k] = Variant(); break;
81  default: break;
82  }
83  }
84  iter = json_object_iter_next(json, iter);
85  }

211  json_t * JsonGenerator::jsonObjectFromValue(const Variant & value)
212  {
213  switch (value.type())
214  {
215  case Variant::TypeInteger: return json_integer(value.toInt64()); break;
216  case Variant::TypeUnsignedInteger: return json_integer(value.toUInt64()); break;
217  case Variant::TypeReal: return json_real(value.toDouble()); break;
218  case Variant::TypeBool: return value.toBool() ? json_true() : json_false(); break;
219  case Variant::TypeString: return json_string(value.toString()); break;
220  case Variant::TypeMap:
221  {
222  json_t * subMapJson = json_object();
223  if (subMapJson)
224  {
225  JsonGenerator::addMapToJson(value.toMap(), subMapJson);
226  return subMapJson;
227  }
228  }
229  break;
230  case Variant::TypeList: return JsonGenerator::jsonObjectFromList(value.toList()); break;
231  default: break;
232  }
233  return json_null();
234  }
